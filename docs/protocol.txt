

einstellungen mit der seriellen schnittstelle
case insensitive



Start := 
  Command '\r'
  ;

Command :=
  "SET" SetCommands
  | "GET" GetCommands
  | "HELLO"                        // greeting command
  ;

SetCommands := 
  "TIMER" Id TimerSetCommands
  | "TEMPSETPOINT"
  | "HUMIDITYSETPOINT"
  | "TIME" Time
  | "OUTPUT" Id Name Enabled
  ;

GetCommands :=
  "TIMERS"
  | "TEMP"
  | "TEMPSETPOINT"
  | "HUMIDITY"
  | "HUMIDITYSETPOINT"
  | "TIME"
  | "OUTPUTS"
  | "VERSION"
  ;

TimerSetCommands :=
  Start End Output
  | ON
  | OFF
  ;


Enabled := 0 | 1
ID    := numeral
Start := Time
End   := Time
Ouput := numeral [ 1..4 ]
Time  := "hh:mm:ss"
