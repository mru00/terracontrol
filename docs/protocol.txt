Start := 
  "SET" SetCommands

  "TIMER" TimerSetCommands
  | "TIMER
  | "HEATING" Value
  | "HUMIDITY" Value
  | "TIME" Time
  ;

GetCommands :=
  "TIMER" ID
  | "HEATING"
  | "HUMIDITY"
  | "TIME"
  ;

TimerSetCommands :=
  "ENABLED"
  | "DISABLED"
  |  Id Start End Output
  ;

Start := Time
End   := Time
Ouput := numeral [ 1..4 ]
